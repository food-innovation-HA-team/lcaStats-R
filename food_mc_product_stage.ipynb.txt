    "# -------------------------------\n",
    "# Monte Carlo Simulation\n",
    "# -------------------------------\n",
    "\n",
    "# Dictionaries to store results\n",
    "total_impacts = pd.DataFrame(index=range(NUM_SIMULATIONS))\n",
    "stage_impacts = {stage: pd.DataFrame(index=range(NUM_SIMULATIONS)) for stage in stages}\n",
    "\n",
    "# Loop over each food item\n",
    "for idx, food in enumerate(food_items):\n",
    "    total_impact = np.zeros(NUM_SIMULATIONS)\n",
    "    \n",
    "    # Loop over each supply chain stage\n",
    "    for stage in stages:\n",
    "        # Get mean and variance\n",
    "        mean = mean_data.at[idx, stage]\n",
    "        variance = variance_data.at[idx, stage]\n",
    "        \n",
    "        # **Adjusted Calculations**\n",
    "        # Calculate GSD\n",
    "        GSD = np.sqrt(variance)\n",
    "        # Calculate sigma\n",
    "        sigma = np.log(GSD)\n",
    "        # Calculate mu\n",
    "        sigma_squared = sigma ** 2\n",
    "        mu = np.log(mean) - (sigma_squared / 2)\n",
    "        \n",
    "        # Generate random samples\n",
    "        samples = np.random.lognormal(mean=mu, sigma=sigma, size=NUM_SIMULATIONS)\n",
    "        \n",
    "        # Store stage impacts\n",
    "        stage_impacts[stage][food] = samples\n",
    "        \n",
    "        # Accumulate total impact\n",
    "        total_impact += samples\n",
    "    \n",
    "    # Store total impacts for the food item\n",
    "    total_impacts[food] = total_impact"
   ]
